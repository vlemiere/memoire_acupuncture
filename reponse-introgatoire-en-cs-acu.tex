\section*{Correspondances réponses à l'interrogatoire et syndromes en MTC \\ extrait du livre de APL}


\subsection*{Les frissons / la fièvre}

\begin{itemize}
\item []Crainte du froid, frissons
\item []Vent-Froid externe (6 niveaux)
\item []Vent-Chaleur externe (4 couches)
\item []Persiste même couvert : \textbf{vide de \textsc{yang}} 
\item []Fièvre l'après midi ou la nuit, bouffées de chaleur : \textbf{vide de \textsc{yin}}
\item []Fièvre légère et constante : \textbf{Chaleur-Humidité}
\end{itemize}

\subsection*{La transpiration}

\subsubsection*{Quantité}
\begin{itemize}
\item []Abondante : \textbf{plénitude de \textsc{yang}} 
\item []Absente : \textbf{vide des \textsc{jin ye}}, plénitude de surface, \textbf{vide de \textsc{yang}}  
\end{itemize}

\subsubsection*{Zone corporelle}
\begin{itemize}
\item []Forte chaleur au niveau de l'estomac : \textbf{Chaleur-Humidité}
\item []Grasse sur le front : \textbf{vide de \textsc{yang}} 
\item []Bras et jambes : \textbf{vide de \textsc{rate / estomac}} 
\item []Main : \textbf{vide de \textsc{qi} du \textsc{poumon}}, problèmes nerveux 
\item []Tout le corps : \textbf{vide de \textsc{qi} du \textsc{poumon}}
\item []Des 5 C\oe{}urs : paumes des mains, plantes des pieds et de la poitrine : \textbf{vide de \textsc{yin}}
\end{itemize}

\subsubsection*{Moment}
\begin{itemize}
\item []En journée : \textbf{vide de \textsc{yang}}
\item []La nuit : \textbf{vide de \textsc{yin}}, \textbf{Chaleur-Humidité}
\end{itemize}

\subsection*{La tête}
Tous les méridiens \textsc{yang} se rencontrent à la tête en apportant le \textsc{yang} aux orifices pour une bonne vue (Bois), une bonne ouïe (Eau), un bon goüt (Terre) et un bon odorat (Métal).

\subsubsection*{Céphalées, description}
\begin{itemize}
\item []Apparition récente : \textbf{Vent-Froid}
\item []Progressive par crises: pathologie interne
\end{itemize}

\subsubsection*{Céphalées, moment}
\begin{itemize}
\item []En journée: \textbf{vide de \textsc{qi}}, \textbf{vide de \textsc{yang}}
\item []En soirée : \textbf{vide de \textsc{yin}}, \textbf{vide de \textsc{sang}}
\end{itemize}

\subsubsection*{Céphalées, localisation}
\begin{itemize}
\item []Nuque : \textbf{\textsc{tai yang}}, invasion Vent-Froid, \textbf{vide de \textsc{rein}}
\item []Frontale et faciale : \textbf{\textsc{yang ming}}, chaleur \textsc{estomac}, \textbf{vide de \textsc{sang}}
\item []Latérale et temporale : \textbf{\textsc{shao yang}}, Vent-Froid, Vent-Chaleur, feu du \textsc{foie - vésicule biliaire}
\item []Vertex : \textbf{\textsc{shao yang}},  \textbf{vide de \textsc{sang} du \textsc{foie}}
\item []Cephalées globales : Vent-Froid externe
\end{itemize}

\subsubsection*{Céphalées, type de douleur}
\begin{itemize}
\item []Lourdeur :Glaires-Humidité
\item []A l'intérieur : \textbf{vide de \textsc{rein}}
\item []Pulsatile : \textbf{élévation du \textsc{yang} du \textsc{foie}}
\item []Comme si on enfonçait un clou : \textbf{stase de \textsc{sang}}
\end{itemize}


\subsubsection*{Céphalées, modalité d'aggravation}
\begin{itemize}
\item []Crainte du vent ou du froid : attaque externe
\item []Aggravées par le froid : Froid
\item []Aggravées par la chaleur : Chaleur
\item []Aggravées par la fatigue, soulagées au repos : \textbf{vide de \textsc{qi}}
\end{itemize}

\subsubsection*{Vertiges}
\begin{itemize}
\item []Avec perte d'équilibre : Vent interne
\item []Avec lourdeur et brouillard : glaire
\item []Legers à la fatigue : \textbf{vide de \textsc{qi}}, malaise vagual chez la femme enceinte
\item []Brusques : plénitude
\item []Apparition progressive : vide
\end{itemize}

\subsection*{Le corps}

\subsubsection*{Douleurs généralisées}
\begin{itemize}
\item []Apparition soudaine : Vent-Froid externe
\item []Avec fatigue : \textbf{vide de \textsc{qi}}, \textbf{vide de \textsc{sang}}
\item []Après l'accouchement, sourdes : \textbf{vide de \textsc{sang}} 
\item []Après l'accouchement, intense : \textbf{stase de \textsc{sang}} 
\item []Dans les muscles avec chaleur : Chaleur de l'\textsc{estomac}
\item []Avec lourdeur : Humidité
\end{itemize}

\subsubsection*{Douleurs articulaires}
\begin{itemize}
\item []Changeantes : Vents
\item []Fixes, intenses : Froid
\item []Fixes avec enflures, engourdissement : Humidité
\end{itemize}

\subsubsection*{Lombalgies}
\begin{itemize}
\item []Incessantes, sourdes : \textbf{vide de \textsc{rein}}
\item []Récentes, intenses, raideur : \textbf{stase de \textsc{sang}} (contractures, entorses)
\item []Intenses aggravées par temps froid et humide
\item []Soulagées par la chaleur : Froid-Humidité
\item []"En barre", impossibilité de bouger au niveau de la taille : \textbf{stase de \textsc{sang}} 
\item []Douleurs du dos qui s'étend aux épaules : atteinte externe
\end{itemize}

\subsubsection*{Engourdissements}
\begin{itemize}
\item []Des bras et jambes : \textbf{vide de \textsc{sang}}
\item []Doigts, coude, bras d'un coté : Vent interne, glaires
\end{itemize}

\subsection*{Le thorax}

\subsubsection*{\textsc{c\oe{}ur}, \textsc{poumon}}
\begin{itemize}
\item []Douleurs thoraciques : \textbf{stase de \textsc{sang} du \textsc{c\oe{}ur}}, \textbf{vide de \textsc{yang}} 
\item []Avec toux : Chaleur du \textsc{poumon}
\end{itemize}

\subsubsection*{Hypocondre - \textsc{foie}, \textsc{vésicule biliaire}}
\begin{itemize}
\item []Distension, plénitude des hypocondre : \textbf{stagnation du \textsc{qi} du \textsc{foie}}
\item []Douleurs intenses des hypocondres : \textbf{stase de \textsc{sang} du \textsc{foie}} 
\end{itemize}

\subsection*{Abdomen}

\subsubsection*{\textsc{foie}, \textsc{intestin}, \textsc{rate}, \textsc{rein}, \textsc{vessie}}
\begin{itemize}
\item []Douleurs épigastriques : rétention de nourriture dans l'Estomac, Chaleur de l'\textsc{estomac}
\item []Douleurs sourdes peu intenses : Froid-Vide de l'\textsc{estomac}
\item []Douleurs soulagées par la prise de nourriture : type vide
\item []Douleurs aggravées par la prise de nourriture : type plénitude
\item []Ballonnement épigastrique : Vide de \textsc{rate}, Humidité
\item []Douleurs bas abdomen : Froid interne, \textbf{stagnation du \textsc{qi} du \textsc{foie}}, \textbf{stagnation de \textsc{sang} du \textsc{foie}}, Chaleur-Humidité, stagnation de \textsc{sang} dans les intestins ou utérus
\item []Douleurs hypogastiques : Chaleur-Humidité de \textsc{vessie}, Feu du \textsc{foie}
\end{itemize}

\subsection*{La nourriture et les saveurs}

\subsubsection*{Renseignent sur l'état de la \textsc{rate} et de l'\textsc{estomac}}
\begin{itemize}
\item []Symptômes soulagées par la prise de nourriture : vide
\item []Symptômes aggravées par la prise de nourriture : plénitude
\item []Manque d'appétit : \textbf{vide de \textsc{qi} de \textsc{rate/estomac}}
\item []Aversion pour la nourriture riche : Chaleur-Humidité du \textsc{foie} et de \textsc{vésicule biliaire}, \textbf{ni du \textsc{qi} de l'\textsc{estomac}}
\item []Gros appétit : Chaleur de l'\textsc{estomac}
\end{itemize}

\subsubsection*{Gout dans la bouche}
\begin{itemize}
\item []Amer : Chaleur du\textsc{foie} (constant), Chaleur du \textsc{c\oe{}ur} (absent si bonne nuit), Chaleur-Humidité dans la \textsc{vésicule biliaire}
\item []Sucré : \textbf{vide de \textsc{rate}}, Chaleur-Humidité
\item []Acide : rétention d'aliments dans l'Estomac, déséquilibre du \textsc{foie} ou l'\textsc{estomac}, Chaleur du\textsc{foie} et de l'\textsc{estomac}
\item []Salé : \textbf{vide de \textsc{rein}}, Froid
\item []Piquant : Chaleur du\textsc{poumon} ou de l'\textsc{estomac}
\item []Perte du sens du goût : \textbf{vide de \textsc{rate/estomac}}
\item []Pâteux : Chaleur-Humidité de \textsc{rate/estomac}
\end{itemize}

\subsubsection*{Nausées, vomissements}
\begin{itemize}
\item []Acide : attaque de l'\textsc{estomac} par le \textsc{foie}
\item []Amers : Chaleur-Humidité du \textsc{foie/vésicule biliaire} 
\item []Clairs aqueux : rétention de liquides, Froid de l'\textsc{estomac}
\item []Peu après mangé : Chaleur
\end{itemize}

\subsection*{Les selles}

\subsubsection*{La constipation}
\begin{itemize}
\item []Constipation aigue, soif, enduit jaune : Chaleur de l'\textsc{estomac}, Chaleur des \textsc{intestins}
\item []Constipation du post partum : \textbf{vide de \textsc{sang}}
\item []Selles petites et sèches : \textbf{stagnation du \textsc{qi} du \textsc{foie}}, Chaleur des \textsc{intestins}, \textbf{vide de \textsc{sang}},  \textbf{vide de \textsc{qi}}
\item []Constipation avec selles sèches sans soif : \textbf{vide de \textsc{yin}}, \textbf{vide de \textsc{yin} du \textsc{rein} ou de l'\textsc{estomac}}
\item []Constipation avec douleurs abdominales : Froid interne, \textbf{vide de \textsc{yang}}
\item []Alternance constipation/diarrhée : \textbf{stagnation du \textsc{qi} du \textsc{foie}} qui envahit la \textsc{rate}, \textbf{vide de \textsc{rate}}, \textbf{vide de \textsc{qi}}, Humidité
\item []Borborygmes et distension : \textbf{stagnation du \textsc{qi} du \textsc{foie}}
\end{itemize}

\subsubsection*{La diarrhée}
\begin{itemize}
\item []Selles avec aliments non digérés : \textbf{vide de \textsc{qi} de \textsc{rate}}
\item []Selles noires très foncées : \textbf{stase de \textsc{sang}}
\item []Chroniques : \textbf{vide de \textsc{yang} de \textsc{rate} et du \textsc{rein}}
\item []Avec douleurs abdominales : Froid interne dans les \textsc{intestins}
\item []Avec mucosités : Humidité dans les \textsc{intestins}
\item []Avec mucosités et sang : Chaleur-Humidité dans les \textsc{intestins}
\end{itemize}

\subsection*{Les urines}

\subsubsection*{Fonction urinaire}
\begin{itemize}
\item []Incontinence : \textbf{vide de \textsc{rein}}, \textbf{vide de \textsc{qi}}
\item []Rétention urinaire : Chaleur-Humidité de \textsc{vessie}
\item []Dysurie : Chaleur-Humidité de \textsc{vessie}, \textbf{stase de \textsc{sang}}, calculs
\item []Brulures mictionnelles : Chaleur-Humidité 
\item []Douleurs mictionnelles : \textbf{vide de \textsc{qi}}
\end{itemize}

\subsubsection*{Couleur}
\begin{itemize}
\item []Claires : vide, Froid du {rein}, Froid de \textsc{vessie}
\item []Foncées : Chaleur
\item []Troubles : Humidité
\end{itemize}

\subsubsection*{Volume}
\begin{itemize}
\item []Volume augmenté : \textbf{vide de \textsc{yang} du \textsc{rein}}
\item []Volume diminué : \textbf{vide de \textsc{yin} du \textsc{rein}}, \textbf{vide de \textsc{qi} du \textsc({poumon}}, \textbf{vide de \textsc{qi} du \textsc({rate}}, Chaleur
\end{itemize}
